\begin{lstlisting}
\documentclass[a4paper]{report}

\usepackage[top=2cm, bottom=2cm, left=2cm, right=2cm]{geometry}

\usepackage{fontspec}
\setmainfont[Scale=1.6]{TH Sarabun New}
\XeTeXlinebreaklocale “th”
\XeTeXlinebreakskip = 0pt plus 1pt
\linespread{1.2}

\usepackage{graphicx}
\usepackage{float}
\usepackage{url}

\begin{document}

\end{document}
\end{lstlisting}
